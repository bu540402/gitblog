/*题意：给出一个骰子，六个面标记数字。放在国际象棋的棋盘里。
 * 没滚动一次，花费着地的那个面标记数字那么多的费用。
 * 给出一个起点与终点。问最小费用是多少。
 * 解法：对于每个点骰子的状态有24种，所以把每个点拆成24种状态。
 * 这样就变成了一个最短路问题。在实现过程中，用map把6个面对应的
 * 数字序列映射成24种状态。两个点之间可转移的状态认为有一条边。
 * 
 */
#include <functional>
#include <algorithm>
#include <iostream>
#include <iomanip>
#include <numeric>
#include <cstring>
#include <cassert>
#include <cstdio>
#include <string>
#include <vector>
#include <bitset>
#include <queue>
#include <stack>
#include <math.h>
#include <ctime>
#include <list>
#include <set>
#include <map>
#include<time.h>
using namespace std;
#define LL long long
#define inf 0x3fffffff
map<LL, int> mm;
int dist[8][8][24];
bool used[8][8][24];
int path[8][8][24][3];
int dir[4][2] =
{
{ 0, 1 },
{ 0, -1 },
{ -1, 0 },
{ 1, 0 } };
int hash = 0;
struct data
{
        int pane[6];
        LL gethash()
        {
            LL ans = 0;
            for (int i = 0; i < 6; i++)
            {
                ans = ans * 1000 + pane[i];
            }
            return ans;
        }

};
int roll(int dir, data &a, data b)
{
    if (dir == 0)
    {
        a.pane[2] = b.pane[3];
        a.pane[3] = b.pane[4];
        a.pane[4] = b.pane[5];
        a.pane[5] = b.pane[2];
    }
    if (dir == 1)
    {
        a.pane[2] = b.pane[5];
        a.pane[3] = b.pane[2];
        a.pane[4] = b.pane[3];
        a.pane[5] = b.pane[4];
    }
    if (dir == 2)
    {
        a.pane[0] = b.pane[4];
        a.pane[1] = b.pane[2];
        a.pane[2] = b.pane[0];
        a.pane[4] = b.pane[1];
    }
    if (dir == 3)
    {
        a.pane[0] = b.pane[2];
        a.pane[1] = b.pane[4];
        a.pane[2] = b.pane[1];
        a.pane[4] = b.pane[0];
    }
}
int check(int x, int y)
{
    if (x < 0 || y < 0 || x > 7 || y > 7)
        return 0;
    return 1;
}
void spfa(int x, int y, data a)
{
    memset(path, -1, sizeof(path));
    queue<int> q;
    queue<data> dq;
    q.push(x);
    q.push(y);
    memset(used, 0, sizeof(used));
    for (int i = 0; i < 8; i++)
        for (int j = 0; j < 8; j++)
            for (int k = 0; k < 24; k++)
            {
                dist[i][j][k] = inf;
            }
    LL tp = a.gethash();
    if (!mm[tp])
        mm[tp] = hash++;
    used[x][y][mm[tp]] = 1;
    dist[x][y][mm[tp]] = a.pane[4];
    q.push(mm[tp]);
    dq.push(a);
    while (!q.empty())
    {
        int ux, uy, ut;
        ux = q.front();
        q.pop();
        uy = q.front();
        q.pop();
        ut = q.front();
        q.pop();
        data u = dq.front();
        dq.pop();
        used[ux][uy][ut] = 0;
        for (int i = 0; i < 4; i++)
        {
            data v = u;
            roll(i, v, u);
            int vx, vy;
            vx = ux + dir[i][0];
            vy = uy + dir[i][1];
            LL tp = v.gethash();
            if (!mm[tp])
                mm[tp] = hash++;
            int vt = mm[tp];
            if (check(vx, vy))
            {
                if (dist[vx][vy][vt] > dist[ux][uy][ut] + v.pane[4])
                {
                    dist[vx][vy][vt] = dist[ux][uy][ut] + v.pane[4];
                    path[vx][vy][vt][0] = ux;
                    path[vx][vy][vt][1] = uy;
                    path[vx][vy][vt][2] = ut;

                    if (!used[vx][vy][vt])
                    {
                        //cout<<vx<<" "<<vy<<" "<<    dist[vx][vy][vt]<<endl;
                        //cout<<ux<<" "<<uy<<" "<<dist[ux][uy][ut]<<endl;;
                        q.push(vx);
                        q.push(vy);
                        q.push(vt);
                        dq.push(v);
                        used[vx][vy][vt] = 1;
                    }
                }
            }
        }
    }
}
vector<string> res;
void getans(int tx, int ty, int tt)
{
    if (tx == -1 || ty == -1 || tt == -1)
        return;
    string s;
    s.push_back('a' + 7 - ty);
    s.push_back('1' + 7 - tx);
    //cout<<s<<endl;
    getans(path[tx][ty][tt][0], path[tx][ty][tt][1], path[tx][ty][tt][2]);
    res.push_back(s);

}
int main()
{
    freopen("data.in", "r", stdin);
    string s, e;
    cin >> s >> e;
    data a;
    int tx, ty, x, y;
    x = s[0] - 'a';
    y = s[1] - '1';
    tx = e[0] - 'a';
    ty = e[1] - '1';
    for (int i = 0; i < 6; i++)
    {
        cin >> a.pane[i];
    }
    spfa(7 - y, 7 - x, a);
    int ans = inf;
    int pos;
    for (int i = 0; i < 24; i++)
    {
        if (ans > dist[7 - ty][7 - tx][i])
        {
            ans = dist[7 - ty][7 - tx][i];
            pos = i;
        }
    }
    cout << ans << endl;
    getans(7 - ty, 7 - tx, pos);
    for (int i = 0; i < res.size(); i++)
    {
        cout << res[i] << " ";
    }
    cout << endl;

}
